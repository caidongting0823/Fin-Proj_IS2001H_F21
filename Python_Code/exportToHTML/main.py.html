# Function Uses in This Python Project #

def countdown(t):
    while t:
        if t >= 0:
            mins, secs = divmod(t, 60)
            timeformat = '{:02d}:{:02d}'.format(mins, secs)
            print("\r" + "Conversion backs in " + timeformat, end=" ", flush=True)
            time.sleep(1)
            t -= 1
        else:
            break
